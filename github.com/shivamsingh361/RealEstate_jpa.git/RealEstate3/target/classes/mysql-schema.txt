CREATE TABLE property(prop_id int(10) PRIMARY KEY AUTO_INCREMENT, prop_address TINYBLOB, prop_price varchar(10), prop_landmark varchar(30), prop_description varchar(80), type varchar(20), user varchar(255));
CREATE TABLE user(login_id varchar(40) PRIMARY KEY, password varchar(30), name varchar(30), address TINYBLOB, phone_no varchar(13), type varchar(20));
CREATE TABLE interest_log(user_id varchar(40) not null, timestamp datetime not null, intrested_city varchar(40), prop_id int(10) not null);
